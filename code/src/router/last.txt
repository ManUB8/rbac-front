export const PrivateBareRoute = () => {
    const { getAuthToken } = useAuth();
    const location = useLocation();
    const navigate = useNavigate();

    if (R.isEmpty(getAuthToken())) return <Navigate to="/login" replace />;

    const titleFromRoute = getRouteNameByPath(location.pathname);

    // ✅ ตรวจจับมือถือ
    const theme = useTheme();
    const isMobile = useMediaQuery(theme.breakpoints.down('md'));

    // ใช้ค่าคงที่เดียวกับใน ModalNavHeader
    const HEADER_H = 64;
    const FOOTER_H = 72;

    return (
        <Box sx={{ bgcolor: 'white' }}>
            <ModalHeaderProvider
                initialTitle={titleFromRoute}
                onClose={() => navigate(-1)}
                renderHeader={(title, actions, onClose) => (
                    <ModalNavHeader
                        title={title}
                        actions={actions}
                        onClose={onClose}
                        isMobile={isMobile}   // ✅ ส่งสถานะมือถือเข้าไป
                    />
                )}
            >
                {/* กันพื้นที่หัว + เท้า (เฉพาะมือถือ) */}
                <Container
                    // sx={{
                    //     bgcolor: 'white',
                    //     mt: `${HEADER_H / 8}`,                 // 64px / 8 = 8 (spacing unit)
                    //     pb: isMobile ? `${FOOTER_H / 8}` : 2,  // 72px / 8 = 9
                    //     marginTop: 8,
                    //     marginBottom:8
                    // }}
                    // sx={{
                    //     bgcolor: 'white',
                    //     mt: HEADER_H / 8,              // 8 → 64px
                    //     pb: isMobile ? FOOTER_H / 8 : 2, // 9 → 72px หรือ 16px
                    // }}
                    sx={{
                        bgcolor: 'white',
                        // กันชน header (64px) และ footer (72px)
                        mt: `${HEADER_H}px`,
                        pb: isMobile ? `${FOOTER_H}px` : 16, // footer หรือ default 16px
                        // ระยะห่างภายใน Outlet
                        py: 2, // 16px ทั้งบนและล่าง
                    }}
                >
                    <Outlet />
                </Container>
            </ModalHeaderProvider>
        </Box>
    );
};
